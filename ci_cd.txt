    testers_string = testers_list.join(", ")

    firebase_app_distribution(
      app: "1:1032756631378:android:10f1b07bb58ce0e44ecc8b",
      firebase_cli_token: ENV["FIREBASE_CLI_TOKEN"],
      android_artifact_type: "APK",
